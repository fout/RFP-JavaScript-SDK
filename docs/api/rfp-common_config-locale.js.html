'use strict';

/** @namespace */
var RFP = window.RFP = window.RFP || {};
/** @namespace */
RFP.Common = RFP.Common || {};

(function() {
    if (RFP.arguments.country) {
        var country = RFP.arguments.country;
        var config = RFP.Common.Locale.Config[country];
        if (config) {
            for (var key in config) {
                RFP.Common.Config[key] = config[key];
            }
        }
    }
})();
