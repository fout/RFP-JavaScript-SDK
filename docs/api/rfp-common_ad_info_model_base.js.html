'use strict';

/** @namespace */
var RFP = window.RFP = window.RFP || {};
/** @namespace */
RFP.Common = RFP.Common || {};

RFP.Common.AdInfoModelBase = (function() {
    var Util = RFP.Common.Util;

    var base_keys = [
        'advertiser_id',
        'ad_id',
        'title',
        'description',
        'optout',
        'optout_click',
        'dat',
        'conv_type',
        'session_id',
        'url_scheme',
        'cta_text'
    ];

    /**
     * @constructor
     * @memberOf RFP.Common
     */
    function AdInfoModelBase(info, adspot_id){
        this.base_info = this.store(info, adspot_id);
    }

    /**
     * @param {object} info - Ad Info Model
     * @param {string} adspot_id - Ad Spot ID
     * @return {object} an object
     */
    AdInfoModelBase.prototype.store = function(info, adspot_id) {
        var result = {};
        Util.forEach(base_keys, function(key) {
            result[key] = info[key];
        });
        result.click_url = RFP.Common.NetworkAd.buildClickRequestUrl(adspot_id, info);

        return result;
    };

    return AdInfoModelBase;
})();
